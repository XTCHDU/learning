int init(int x)
{

    if (!(ch>='A'&&ch<='O')||!(n>=1&&n<=15)) return 1;

    x++;
    if (a[n][m]!=0) {return 1;}
    a[n][m]=x;
    check(n,m);

}
int search(int now,int x,int y,int k,int count)

int search1(int now,int x,int y,int k,int count,int r,int flag)
{
    x+=dx[k];
    y+=dy[k];
    if (a[x][y]==now) {if (r==1) flag=1;search1(now,x,y,k,count+1,r,flag);}
     else 
     if (a[x][y]==0 && r==0)
     {
                    search1(now,x,y,k,count,r+1,flag);
                    }
                    else {
                             if (a[x][y]!=0&&a[x][y]!=now) zu[k]=1;
                             if (r==1&&flag) kong[k]=1;
                             f[k]=count;return 0;}
}
